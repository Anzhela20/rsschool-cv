public class Bubble {
	public static void main(String[] args) {

		int[] l = { 2, 4, 6, 23, 12, 1, 7, 0, 13, 17, 27 };

		boolean isSorted = false;

		while (!isSorted) {
			isSorted = true;
			for (int i = 1; i < l.length; i++) {
				if (l[i] < l[i - 1]) {
					int d;
					d = l[i];
					l[i] = l[i - 1];
					l[i - 1] = d;
					isSorted = false;
				}
			}
		}
		System.out.println(Arrays.toString(l));
	}
}
